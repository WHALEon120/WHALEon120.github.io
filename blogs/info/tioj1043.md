# 我的AC CODE
``` cpp
#include<bits/stdc++.h>
#define int long long 
using namespace std;
int dp[10010]={1};
signed main() {
    int p,c,n;
    while (cin>>n>>p&&n&&p) {
        for (int i=1; i<=p; i++) {
            dp[i]=0;
        }
        for (int i=0; i<n; i++) {
            cin>>c;
            for (int j=c; j<=p; j++) {
                dp[j]+=dp[j-c];
            }
        }
        cout<<dp[p]<<'\n';
    }
}
```

#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	ios::sync_with_stdio(false);
	cin.tie(0);
	long long dp[10001], w[100];
	int n, p, i, j;
	dp[0]=1;
	while(cin >> n >> p and n !=0){
		for(i=1;i<=p;i++){
			dp[i]=0;
		}
		for(i=0;i<n;i++){
			cin >> w[i];
		}
		for(i=1;i<=p;i++){
			for(j=0;j<n;j++){
				if(w[j]<=i){
					dp[i] += dp[i-w[j]];
				}
			}
		}
		cout << dp[p] << '\n';
	}
	return 0;
}
