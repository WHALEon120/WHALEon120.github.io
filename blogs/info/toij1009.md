# 我的AC CODE  
``` cpp
#include <bits/stdc++.h>
using namespace std;
int read_input() {
    int h, m, s;
    scanf("%d:%d:%d", &h, &m, &s);
    return (h*60+m)*60+s;
}
void output(int sec) {
    printf("%02d:%02d:%02d\n", sec/3600, sec/60%60, sec%60);
}
int main() {
    int A = read_input();
    int B = read_input();
    //tomorrow
    if (A >= B) 
        B += 86400;
    
    output(B-A);
    return 0;
}
```
