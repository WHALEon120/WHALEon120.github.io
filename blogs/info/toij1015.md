    ll n,m;
    while(cin>>n>>m&&n+m!=0){
        if(n>m) swap(n,m);
        ll r = n*(n+1)*(3*m-n+1)/6;
        cout<<r<<'\n';
    }

#include <bits/stdc++.h>
using namespace std;

long long min(long long a, long long int b){
	if(a>b){
	return b;
	}
	else{
		return a;
	}
}
int main() {
	// your code goes here
	ios::sync_with_stdio(0);
	cin.tie(0);
	long long a, b, r;
	while(cin >> a >> b){
		r=0;
		if(a !=0 and b != 0){
		for(long long i=0;i<min(a,b);i++){
			r += (a-i)*(b-i);
		}
		cout << r <<'\n';
		}
	
	else{
		break;
	}
	}
	return 0;
}
