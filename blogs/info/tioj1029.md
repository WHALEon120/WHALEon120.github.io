# 我的AC CODE
``` cpp
#include <bits/stdc++.h>
using namespace std;
int n, s, a[1005][1005];
int main() {
  ios::sync_with_stdio(false);
  cin.tie(0);
    cin >> n;
    for (int i = 2; i < n+2; i++){
        cin >> a[i][i];
        s += a[i][i];
        for (int j = i-1; j >= 2; j--){
            a[j][i] = max(a[j][j] + min(a[j+1][i-1], a[j+2][i]),
                          a[i][i] + min(a[j][i-2], a[j+1][i-1]));
        }
    }
    cout << a[2][n+1] << " " << s - a[2][n+1] << "\n";
}

```
